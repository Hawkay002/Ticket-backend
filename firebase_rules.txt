rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Only allow access if the user matches the document ID
    // This stops User A from hacking User B's data
    match /ticket_events_data/{userId}/{document=**} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
  }
}
